Integración con Google Sheets y Drive

URL Google Apps Script

Web App URL para envío automático a Google Sheets. Cada boleta guardada se registra como fila nueva.

ID carpeta Google Drive (para fotos)

ID de la carpeta de Google Drive donde se guardarán las fotos. Lo encuentras en la URL de tu carpeta: drive.google.com/drive/folders/ESTE_ES_EL_ID. Si lo dejas vacío, las fotos se subirán a la raíz de Drive.

Código Apps Script

Pega en script.google.com → nuevo proyecto → desplegar como Web App (acceso: cualquier usuario).

// ── Belcorp Boletas — Apps Script ────────────────────────── // Desplegar como Web App: // Ejecutar como: Yo (tu cuenta) // Acceso: Cualquier usuario (o anónimo) var FOLDER_ID = ''; // ← ID carpeta Drive (opcional) function doPost(e) { try { var data = JSON.parse(e.postData.contents); // 1. SUBIR FOTO A DRIVE var urlFoto = ''; if (data.imageBase64 && data.fileName) { var folderId = data.folderId || FOLDER_ID; var folder = folderId ? DriveApp.getFolderById(folderId) : DriveApp.getRootFolder(); var blob = Utilities.newBlob( Utilities.base64Decode(data.imageBase64), data.mimeType || 'image/jpeg', data.fileName ); var file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); urlFoto = 'https://drive.google.com/file/d/' + file.getId() + '/view'; } // 2. REGISTRAR EN SHEETS var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow(['Campaña','ID Consultora','Nº Pedido','Zona','Fecha','URL Foto']); } sheet.appendRow([ data.campana, data.consultora, data.pedido, data.zona, data.fecha, urlFoto ]); return ContentService .createTextOutput(JSON.stringify({ ok: true, urlFoto: urlFoto })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } }
